#include <Arduino.h>

const int ENA_PIN = 27; // the Arduino pin connected to the EN1 pin L298N
const int IN1_PIN = 26; // the Arduino pin connected to the IN1 pin L298N
const int IN2_PIN = 25; // the Arduino pin connected to the IN2 pin L298N
// the setup function runs once when you press reset or power the board
void setup() {
  // initialize digital pins as outputs.
  pinMode(ENA_PIN, OUTPUT);
  pinMode(IN1_PIN, OUTPUT);
  pinMode(IN2_PIN, OUTPUT);
  digitalWrite(ENA_PIN, HIGH);
}
// the loop function runs over and over again forever
void loop() {
  // extend the actuator
  Serial.println("Extend...");
  digitalWrite(IN1_PIN, HIGH);
  digitalWrite(IN2_PIN, LOW);
  //delay(1000); // actuator will stop extending automatically when reaching the limit
  
  // Serial.println("Retract...");
  // digitalWrite(IN1_PIN, LOW);
  // digitalWrite(IN2_PIN, HIGH);
  // delay(2000); // actuator will stop retracting automatically when reaching the limit
}
